import { Block } from './Block'

export function padNum(num) {
  return num > 9 ? num : `0${num}`
}

  title='Components/Block'

  parameters={{ badges: [BADGE.EXPERIMENTAL] }}

Block is a layout component for common block styles.

## Inline block elements

Pass A truthy `inline` prop to create an block element with `display: inline-block`:

<Block backgroundColor='primary-200' className='rounded'>
  {[1, 2, 3].map((key) => (
    <Block
      inline
      key={key}
      className='bg-primary-600 text-inverse font-mono font-bold px-3 py-2 rounded'
    >
      #{padNum(key)}
    </Block>
  ))}
</Block>
