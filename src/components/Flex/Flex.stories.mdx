export function padNum(num) {
  return num > 9 ? num : `0${num}`
}

Flex is a layout component with shorthand properties `align`, `direction`, `justify`, and
`wrap` for common flex styles.

## Flex direction

Override the default direction of `row` with the `direction` prop:

<Flex direction='row-reverse' gap={4} backgroundColor='primary-200' className='rounded'>
  {[1, 2, 3].map((key) => (
    <div
      key={key}
      className='bg-primary-600 text-inverse font-mono font-bold px-6 py-4 rounded'
    >
      #0{key}
    </div>
  ))}
</Flex>

## Justify and align

Set the flex items justification and alignment with the `justify` and `align` props:

<Flex
  justify='center'
  align='center'
  gap={4}
  backgroundColor='primary-200'
  className='rounded'
>
  {[1, 2, 3].map((key) => (
    <div
      key={key}
      className='bg-primary-600 text-inverse font-mono font-bold px-6 py-4 rounded'
    >
      #0{key}
    </div>
  ))}
</Flex>

## Flex wrap

Override the flex wrap behavior with the `wrap` props:

<Flex wrap='wrap-reverse' gap={4} backgroundColor='primary-200' className='rounded'>
  {Array(12)
    .fill(0)
    .map((_, key) => (
      // eslint-disable-next-line react/no-array-index-key
      <div
        key={key}
        className='bg-primary-600 text-inverse font-mono font-bold px-6 py-4 rounded'
      >
        #{padNum(key)}
      </div>
    ))}
</Flex>
